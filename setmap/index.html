        Problem:
        You are given a string (STR) of length N, consisting of only the lower case English alphabet. Your task is to 
        remove all the duplicate occurrences of characters in the string.
        
        Input:
        abcadeecfb
        
        Output:
        abcdef
        

            let s="abcadeeggcfb";
            let unique =new Set();
            for(let i=0;i< s.length;i++){
                unique.add(s[i]);
            }
            let result="";
            for(let key of unique.keys()){
                result=result+key;
            }
            console.log(result);
        

        Problem:
        You are given a string (STR) of length N, you have to print the count of all alphabet.(using maps)

        Input:
        abcadeecfb

        Output:
        a=2
        b=2
        c=2
        d=1
        e=2
        f=1
        

            let myMap= new Map();
            let str="abcadeecfb";
            for(let i=0;i< str.length;i++){
            if(myMap.has(str[i])){
                let previousCount = myMap.get(str[i]);
                myMap.set(str[i],previousCount+1);
            }
            else{
                myMap.set(str[i],1);
            }
            }
            for(let [key,value] of myMap.entries()){
            console.log(key+" ="+value);
