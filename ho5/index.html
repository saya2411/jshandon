        const countCharacters = (S) => 
        {
          let B=[];
          let countOfA=0, countOfD=0;
            for (let i=0;i< S.length;i++){
              if (S[i] === "A"){
                countOfA++;
              }
              else if (S[i] === "D"){
                countOfD++;
              }
            }
            B.push(countOfA, countOfD);
            return B;
            
        };

        var Game_Winner = (S) => 
        {
        let countA=0,countD=0;
        for(let i=0;i< S.length;i++){
            if(S[i]==="A")
                countA=countA+1;
            else if(S[i]==="D")
                countD=countD+1;
        }
        if (countA > countD)
            return "Aditya";
        
        else if (countA < countD)
